// ============================================================================= // Supabase Edge Function: "trading-bot" (MEHRWERTE-VERSION) // Läuft auf einem Zeitplan (Cron), handelt NUR auf dem Paper-Trading-Konto. // // NEU in dieser Version: // - Handelt MEHRERE Werte (Aktien UND/ODER Krypto) aus einer Liste. // - Gemeinsamer Kapitaltopf über alle Werte (Risiko bleibt gedeckelt). // - Krypto (Symbol mit "/", z.B. BTC/USD) wird automatisch erkannt: // * läuft 24/7 (keine Markt-offen-Prüfung) // * eigener Krypto-Datenfeed // ============================================================================= // // KONFIGURATION in Tabelle bot_config (id=1): // symbols = Komma-Liste, z.B. "BTC/USD,ETH/USD,SOL/USD,WIF/USD" // (Aktien gehen weiter, z.B. "AAPL,MSFT") // Hinweis: das alte Feld "symbol" wird weiter unterstützt, falls "symbols" leer ist. // // KILL-SWITCH: bot_config.enabled = false // ============================================================================= import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; // ---- Alpaca Endpunkte (fest auf PAPER) ------------------------------------ const TRADING_BASE = "https://paper-api.alpaca.markets"; const DATA_BASE = "https://data.alpaca.markets"; // ---- Supabase-Client ------------------------------------------------------- const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); // ---- Secrets --------------------------------------------------------------- const ALPACA_KEY = Deno.env.get("ALPACA_KEY")!; const ALPACA_SECRET = Deno.env.get("ALPACA_SECRET")!; // ---- Typen ----------------------------------------------------------------- interface BotConfig { enabled: boolean; symbol: string; // alt (Einzelwert) - Fallback symbols: string | null; // neu: Komma-Liste mehrerer Werte ma_short: number; ma_long: number; order_notional: number; // Kaufbetrag pro Order (je Wert) max_position_notional: number; // Obergrenze je Einzelposition daily_loss_limit: number; // Tagesverlustlimit (gesamtes Konto) capital_pool: number; // GEMEINSAMER Topf über alle Werte stop_loss_pct: number; tax_reserve: number; tax_rate: number; } // ---- Krypto erkennen ------------------------------------------------------- // Alpaca-Kryptosymbole enthalten einen Schrägstrich, z.B. "BTC/USD". const istKrypto = (sym: string) => sym.includes("/"); // ---- Alpaca HTTP-Helfer ---------------------------------------------------- async function alpaca( base: string, method: string, path: string, body?: unknown ): Promise { const res = await fetch(base + path, { method, headers: { "APCA-API-KEY-ID": ALPACA_KEY, "APCA-API-SECRET-KEY": ALPACA_SECRET, "Content-Type": "application/json", }, body: body ? JSON.stringify(body) : undefined, }); const txt = await res.text(); let data: unknown; try { data = txt ? JSON.parse(txt) : {}; } catch { data = { raw: txt }; } if (!res.ok) { const msg = (data as Record).message || txt; throw new Error(`Alpaca ${res.status}: ${msg}`); } return data; } // ---- Indikatoren ----------------------------------------------------------- function sma(s: number[], end: number, len: number): number | null { if (end + 1 < len) return null; let x = 0; for (let k = end - len + 1; k <= end; k++) x += s[k]; return x / len; } function emaArr(s: number[], len: number): number[] { const k = 2 / (len + 1); const o: number[] = []; let p = 0; for (let i = 0; i < s.length; i++) { p = i === 0 ? s[i] : s[i] * k + p * (1 - k); o.push(p); } return o; } function rsiVal(s: number[], end: number, len: number): number | null { if (end < len) return null; let g = 0, l = 0; for (let i = end - len + 1; i <= end; i++) { const c = s[i] - s[i - 1]; g += Math.max(0, c); l += Math.max(0, -c); } g /= len; l /= len; return l === 0 ? 100 : 100 - 100 / (1 + g / l); } function bollinger(s: number[], end: number, len: number, m: number) { if (end + 1 < len) return null; let x = 0; for (let k = end - len + 1; k <= end; k++) x += s[k]; const me = x / len; let v = 0; for (let k = end - len + 1; k <= end; k++) v += (s[k] - me) ** 2; const sd = Math.sqrt(v / len); return { up: me + m * sd, lo: me - m * sd }; } function committee(prices: number[], maS: number, maL: number) { const i = prices.length - 1; const price = prices[i]; const e12 = emaArr(prices, 12), e26 = emaArr(prices, 26); const macdArr = prices.map((_, k) => e12[k] - e26[k]); const signalArr = emaArr(macdArr, 9); const bb = bollinger(prices, i, 20, 2); const r = rsiVal(prices, i, 14); const maShortVal = sma(prices, i, maS); const maLongVal = sma(prices, i, maL); const vMa = (maShortVal != null && maLongVal != null) ? (maShortVal > maLongVal ? 1 : -1) : 0; const vRsi = r == null ? 0 : (r < 30 ? 1 : (r > 70 ? -1 : 0)); const vMacd = macdArr[i] > signalArr[i] ? 1 : (macdArr[i] < signalArr[i] ? -1 : 0); const vBb = bb ? (price < bb.lo ? 1 : (price > bb.up ? -1 : 0)) : 0; const score = vMa + vRsi + vMacd + vBb; return { price, score }; } // ---- Protokollierung ------------------------------------------------------- async function logTrade(data: Record) { await supabase.from("bot_trades").insert(data); } async function logStatus(msg: string) { await supabase.from("bot_log").insert({ message: msg }); } // ---- Kursdaten holen (Aktie ODER Krypto) ----------------------------------- async function holeKurse(symbol: string, maxBars: number): Promise { if (istKrypto(symbol)) { // Krypto-Feed: Stundenbalken (1Hour) damit genug historische Daten vorhanden sind const path = `/v1beta3/crypto/us/bars?symbols=${encodeURIComponent(symbol)}` + `&timeframe=1Hour&limit=${maxBars}`; const d = await alpaca(DATA_BASE, "GET", path) as Record; const map = (d.bars || {}) as Record[]>; const arr = map[symbol] || []; return arr.map((b) => b.c).filter((n) => typeof n === "number"); } else { // Aktien-Feed (kostenloser IEX-Feed) const path = `/v2/stocks/${encodeURIComponent(symbol)}/bars` + `?timeframe=1Day&limit=${maxBars}&feed=iex&adjustment=raw`; const d = await alpaca(DATA_BASE, "GET", path) as Record; const bars = (d.bars || []) as Record[]; return bars.map((b) => b.c).filter((n) => typeof n === "number"); } } // ---- Position eines Symbols holen ------------------------------------------ async function holePosition(symbol: string): Promise | null> { try { // Bei Krypto braucht der Pfad das Symbol ohne "/" (z.B. BTCUSD) const pfadSym = istKrypto(symbol) ? symbol.replace("/", "") : symbol; return await alpaca( TRADING_BASE, "GET", `/v2/positions/${encodeURIComponent(pfadSym)}` ) as Record; } catch { return null; } } // ---- Markt offen? (nur für Aktien relevant) -------------------------------- async function aktienMarktOffen(): Promise { const clock = await alpaca(TRADING_BASE, "GET", "/v2/clock") as Record; return clock.is_open === true; } // ---- Hauptlogik ------------------------------------------------------------ async function runBotTick(): Promise> { // 1) Konfiguration lesen const { data: cfgRow, error: cfgErr } = await supabase .from("bot_config").select("*").eq("id", 1).single(); if (cfgErr || !cfgRow) throw new Error("bot_config nicht gefunden: " + cfgErr?.message); const cfg = cfgRow as BotConfig; // 2) Kill-Switch if (!cfg.enabled) { return { status: "paused", reason: "Bot in bot_config deaktiviert." }; } // 3) Symbol-Liste aufbauen (neu: symbols; Fallback: altes symbol) const liste = (cfg.symbols && cfg.symbols.trim().length > 0) ? cfg.symbols.split(",").map((s) => s.trim()).filter(Boolean) : [cfg.symbol]; if (liste.length === 0) { return { status: "warte", reason: "Keine Symbole in bot_config." }; } // 4) Konto / Tagesverlustlimit (gilt fürs GESAMTE Konto) const account = await alpaca(TRADING_BASE, "GET", "/v2/account") as Record; const dailyPnL = parseFloat(account.equity) - parseFloat(account.last_equity); if (dailyPnL <= -Math.abs(cfg.daily_loss_limit)) { await logStatus(`Tagesverlustlimit erreicht (${dailyPnL.toFixed(2)}€) – kein Handel.`); return { status: "limit", dailyPnL }; } // 5) Aktien-Markt einmal prüfen (für Krypto egal) let aktienOffen: boolean | null = null; // null = noch nicht abgefragt // Laufende Topf-Größe (wird über die Werte hinweg verkleinert/vergrößert) let topf = cfg.capital_pool; let rueckl = cfg.tax_reserve ?? 0; // Steuersatz: Krypto = 50 % (pers. Einkommensteuer + Soli), Aktien = tax_rate aus bot_config // Wird pro Symbol beim Verkauf neu bestimmt (siehe unten) let topfDirty = false; // müssen wir den Topf am Ende speichern? const ergebnisse: Record[] = []; // 6) Jeden Wert der Reihe nach abarbeiten for (const symbol of liste) { try { const krypto = istKrypto(symbol); const prices = await holeKurse(symbol, 120); if (prices.length < cfg.ma_long + 5) { ergebnisse.push({ symbol, status: "warte", bars: prices.length }); continue; } const c = committee(prices, cfg.ma_short, cfg.ma_long); const position = await holePosition(symbol); const invested = !!position; const posValue = invested ? parseFloat(position!.market_value) : 0; const buySig = c.score >= 2 && !invested; const sellSig = c.score <= -2 && invested; await logStatus( `${symbol} | Kurs ${c.price.toFixed(2)} | Score ${c.score > 0 ? "+" : ""}${c.score}` + ` | Position ${invested ? posValue.toFixed(0) + "€" : "keine"}` + ` | Topf ${topf.toFixed(0)}€` ); // Aktien: Markt muss offen sein. Krypto: immer offen. if ((buySig || sellSig) && !krypto) { if (aktienOffen === null) aktienOffen = await aktienMarktOffen(); if (!aktienOffen) { ergebnisse.push({ symbol, status: "market_closed", score: c.score }); continue; } } // ---- KAUFEN ---- if (buySig) { const freiImTopf = topf - posValue; if (freiImTopf <= 1) { await logStatus(`Kauf ${symbol} übersprungen: Topf ausgeschöpft (${topf.toFixed(0)}€).`); ergebnisse.push({ symbol, status: "pool_voll" }); continue; } const kaufBetrag = Math.min(cfg.order_notional, freiImTopf); if (posValue + kaufBetrag > cfg.max_position_notional) { ergebnisse.push({ symbol, status: "limit_position" }); continue; } const orderBody: Record = { symbol, notional: Math.floor(kaufBetrag * 100) / 100, side: "buy", type: "market", time_in_force: krypto ? "gtc" : "day", // Krypto: gtc, Aktie: day }; const order = await alpaca(TRADING_BASE, "POST", "/v2/orders", orderBody) as Record; await logTrade({ symbol, side: "buy", notional: kaufBetrag, price: c.price, score: c.score, order_id: order.id, mode: "paper", }); await logStatus(`KAUF ${symbol}: ${kaufBetrag.toFixed(0)}€ (Topf war ${topf.toFixed(0)}€).`); topf -= kaufBetrag; // Topf für die nächsten Werte verkleinern topfDirty = true; ergebnisse.push({ symbol, status: "buy", betrag: kaufBetrag, price: c.price }); continue; } // ---- VERKAUFEN ---- if (sellSig) { const einstand = parseFloat(position!.cost_basis || "0"); const erloes = posValue; const gewinn = erloes - einstand; const pfadSym = krypto ? symbol.replace("/", "") : symbol; await alpaca(TRADING_BASE, "DELETE", `/v2/positions/${encodeURIComponent(pfadSym)}`); // Steuersatz: Krypto 50 %, Aktien tax_rate aus bot_config (Standard 26,375 %) const satz = krypto ? 0.50 : (cfg.tax_rate ?? 26.375) / 100; let steuer = 0, nettoGewinn = gewinn; if (gewinn > 0) { steuer = gewinn * satz; nettoGewinn = gewinn - steuer; } topf = Math.max(0, topf + nettoGewinn); rueckl = rueckl + steuer; topfDirty = true; await logTrade({ symbol, side: "sell", notional: erloes, price: c.price, score: c.score, order_id: null, mode: "paper", }); await logStatus( `VERKAUF ${symbol}: Erlös ${erloes.toFixed(0)}€, ` + `G/V ${gewinn >= 0 ? "+" : ""}${gewinn.toFixed(0)}€` + (steuer > 0 ? `, Steuer +${steuer.toFixed(2)}€ (ges. ${rueckl.toFixed(2)}€)` : ``) + ` → Topf neu ${topf.toFixed(0)}€.` ); ergebnisse.push({ symbol, status: "sell", gewinn, steuer }); continue; } ergebnisse.push({ symbol, status: "hold", score: c.score, invested }); } catch (e) { const msg = e instanceof Error ? e.message : String(e); await logStatus(`FEHLER bei ${symbol}: ${msg}`); ergebnisse.push({ symbol, status: "error", message: msg }); } } // 7) Topf & Rückstellung einmal speichern, falls sich etwas geändert hat if (topfDirty) { await supabase.from("bot_config") .update({ capital_pool: topf, tax_reserve: rueckl }) .eq("id", 1); } return { status: "ok", dailyPnL, topf, tax_reserve: rueckl, werte: ergebnisse }; } // ---- HTTP-Einstiegspunkt --------------------------------------------------- Deno.serve(async (req) => { if (req.method === "OPTIONS") { return new Response("ok", { headers: { "Access-Control-Allow-Origin": "*" } }); } try { const result = await runBotTick(); return new Response(JSON.stringify(result), { headers: { "Content-Type": "application/json" }, }); } catch (e) { const msg = e instanceof Error ? e.message : String(e); return new Response(JSON.stringify({ error: msg }), { status: 500, headers: { "Content-Type": "application/json" }, }); } });
