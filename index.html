// ============================================================================= // Supabase Edge Function: "trading-bot" // Läuft auf einem Zeitplan (Cron), handelt NUR auf dem Paper-Trading-Konto. // ============================================================================= // // BENÖTIGTE SUPABASE SECRETS (einmalig setzen): // supabase secrets set ALPACA_KEY=DEIN_PAPER_KEY // supabase secrets set ALPACA_SECRET=DEIN_PAPER_SECRET // // DEPLOYEN: // supabase functions deploy trading-bot // // ZEITPLAN EINRICHTEN (Supabase Dashboard): // Edge Functions → trading-bot → Schedule → z.B. "* * * * *" (jede Minute) // ODER: per SQL (siehe bot-tables.sql) // // KILL-SWITCH: // In der Tabelle bot_config den Wert enabled = false setzen. // Nächster Takt: Bot macht nichts, schreibt "paused" ins Protokoll. // ============================================================================= import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; // ---- Alpaca Endpunkte (fest auf PAPER) ------------------------------------ const TRADING_BASE = "https://paper-api.alpaca.markets"; const DATA_BASE = "https://data.alpaca.markets"; // ---- Supabase-Client (Zugangsdaten automatisch injiziert) ----------------- const supabase = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); // ---- Secrets --------------------------------------------------------------- const ALPACA_KEY = Deno.env.get("ALPACA_KEY")!; const ALPACA_SECRET = Deno.env.get("ALPACA_SECRET")!; // ---- Typen ----------------------------------------------------------------- interface BotConfig { enabled: boolean; symbol: string; ma_short: number; ma_long: number; order_notional: number; max_position_notional: number; daily_loss_limit: number; capital_pool: number; // Gesamttopf: nur dieses Kapital (+ Gewinne) wird investiert stop_loss_pct: number; // Stop-Loss in Prozent (z.B. 8 = 8%) } // ---- Alpaca HTTP-Helfer ---------------------------------------------------- async function alpaca( base: string, method: string, path: string, body?: unknown ): Promise { const res = await fetch(base + path, { method, headers: { "APCA-API-KEY-ID": ALPACA_KEY, "APCA-API-SECRET-KEY": ALPACA_SECRET, "Content-Type": "application/json", }, body: body ? JSON.stringify(body) : undefined, }); const txt = await res.text(); let data: unknown; try { data = txt ? JSON.parse(txt) : {}; } catch { data = { raw: txt }; } if (!res.ok) { const msg = (data as Record).message || txt; throw new Error(`Alpaca ${res.status}: ${msg}`); } return data; } // ---- Indikatoren ----------------------------------------------------------- function sma(s: number[], end: number, len: number): number | null { if (end + 1 < len) return null; let x = 0; for (let k = end - len + 1; k <= end; k++) x += s[k]; return x / len; } function emaArr(s: number[], len: number): number[] { const k = 2 / (len + 1); const o: number[] = []; let p = 0; for (let i = 0; i < s.length; i++) { p = i === 0 ? s[i] : s[i] * k + p * (1 - k); o.push(p); } return o; } function rsiVal(s: number[], end: number, len: number): number | null { if (end < len) return null; let g = 0, l = 0; for (let i = end - len + 1; i <= end; i++) { const c = s[i] - s[i - 1]; g += Math.max(0, c); l += Math.max(0, -c); } g /= len; l /= len; return l === 0 ? 100 : 100 - 100 / (1 + g / l); } function bollinger(s: number[], end: number, len: number, m: number) { if (end + 1 < len) return null; let x = 0; for (let k = end - len + 1; k <= end; k++) x += s[k]; const me = x / len; let v = 0; for (let k = end - len + 1; k <= end; k++) v += (s[k] - me) ** 2; const sd = Math.sqrt(v / len); return { up: me + m * sd, lo: me - m * sd }; } function committee(prices: number[], maS: number, maL: number) { const i = prices.length - 1; const price = prices[i]; const e12 = emaArr(prices, 12), e26 = emaArr(prices, 26); const macdArr = prices.map((_, k) => e12[k] - e26[k]); const signalArr = emaArr(macdArr, 9); const bb = bollinger(prices, i, 20, 2); const r = rsiVal(prices, i, 14); const maShortVal = sma(prices, i, maS); const maLongVal = sma(prices, i, maL); const vMa = (maShortVal != null && maLongVal != null) ? (maShortVal > maLongVal ? 1 : -1) : 0; const vRsi = r == null ? 0 : (r < 30 ? 1 : (r > 70 ? -1 : 0)); const vMacd = macdArr[i] > signalArr[i] ? 1 : (macdArr[i] < signalArr[i] ? -1 : 0); const vBb = bb ? (price < bb.lo ? 1 : (price > bb.up ? -1 : 0)) : 0; const score = vMa + vRsi + vMacd + vBb; return { price, score, votes: { ma: vMa, rsi: vRsi, macd: vMacd, bb: vBb, rsiVal: r } }; } // ---- Protokollierung in Supabase ------------------------------------------- async function logTrade(data: Record) { await supabase.from("bot_trades").insert(data); } async function logStatus(msg: string) { await supabase.from("bot_log").insert({ message: msg }); } // ---- Markt offen? ---------------------------------------------------------- async function marketIsOpen(): Promise { const clock = await alpaca(TRADING_BASE, "GET", "/v2/clock") as Record; return clock.is_open === true; } // ---- Hauptlogik ------------------------------------------------------------ async function runBotTick(): Promise> { // 1) Konfiguration aus Datenbank lesen const { data: cfgRow, error: cfgErr } = await supabase .from("bot_config").select("*").eq("id", 1).single(); if (cfgErr || !cfgRow) throw new Error("bot_config nicht gefunden: " + cfgErr?.message); const cfg = cfgRow as BotConfig; // 2) Kill-Switch prüfen if (!cfg.enabled) { return { status: "paused", reason: "Bot in bot_config deaktiviert." }; } // 3) Kursdaten holen (Alpaca Data API, kostenloser IEX-Feed) const barPath = `/v2/stocks/${encodeURIComponent(cfg.symbol)}/bars` + `?timeframe=1Day&limit=120&feed=iex&adjustment=raw`; const barData = await alpaca(DATA_BASE, "GET", barPath) as Record; const bars = (barData.bars || []) as Record[]; const prices = bars.map((b) => b.c).filter((n) => typeof n === "number"); if (prices.length < cfg.ma_long + 5) { return { status: "warte", reason: `Nur ${prices.length} Bars vorhanden.` }; } // 4) Indikatoren berechnen const c = committee(prices, cfg.ma_short, cfg.ma_long); // 5) Alpaca-Konto prüfen (Schutzschranken) const account = await alpaca(TRADING_BASE, "GET", "/v2/account") as Record; const equity = parseFloat(account.equity); const lastEquity = parseFloat(account.last_equity); const dailyPnL = equity - lastEquity; if (dailyPnL <= -Math.abs(cfg.daily_loss_limit)) { await logStatus(`Tagesverlustlimit erreicht (${dailyPnL.toFixed(2)}€) – kein Handel.`); return { status: "limit", dailyPnL }; } // 6) Aktuelle Position prüfen let position: Record | null = null; try { position = await alpaca( TRADING_BASE, "GET", `/v2/positions/${encodeURIComponent(cfg.symbol)}` ) as Record; } catch { /* keine Position vorhanden */ } const invested = !!position; const posValue = invested ? parseFloat(position!.market_value) : 0; const buySig = c.score >= 2 && !invested; const sellSig = c.score <= -2 && invested; await logStatus( `${cfg.symbol} | Kurs ${c.price.toFixed(2)} | Score ${c.score > 0 ? "+" : ""}${c.score}` + ` | Position ${invested ? posValue.toFixed(0) + "€" : "keine"} | Tag-PnL ${dailyPnL.toFixed(2)}€` ); // 7) Markt offen? (bei Aktien nötig) if (buySig || sellSig) { const open = await marketIsOpen(); if (!open) { return { status: "market_closed", score: c.score, signal: buySig ? "buy" : "sell" }; } } // 8) Kaufen if (buySig) { // --- Kapitaltopf-Prüfung: nie mehr als capital_pool im Markt --- // Verfügbarer Spielraum = Topf minus bereits investierter Positionswert const freiImTopf = cfg.capital_pool - posValue; if (freiImTopf <= 1) { await logStatus( `Kauf übersprungen: Topf ausgeschöpft ` + `(Topf ${cfg.capital_pool.toFixed(0)}€, investiert ${posValue.toFixed(0)}€).` ); return { status: "pool_voll", capital_pool: cfg.capital_pool, invested: posValue }; } // Kaufbetrag = kleinerer Wert aus gewünschtem Notional und freiem Topf const kaufBetrag = Math.min(cfg.order_notional, freiImTopf); // zusätzliche Einzelpositions-Obergrenze (bestehende Schranke) if (posValue + kaufBetrag > cfg.max_position_notional) { return { status: "limit", reason: "Max. Positionsgröße würde überschritten." }; } const order = await alpaca(TRADING_BASE, "POST", "/v2/orders", { symbol: cfg.symbol, notional: Math.floor(kaufBetrag * 100) / 100, side: "buy", type: "market", time_in_force: "day", }) as Record; await logTrade({ symbol: cfg.symbol, side: "buy", notional: kaufBetrag, price: c.price, score: c.score, order_id: order.id, mode: "paper", }); await logStatus( `KAUF ${cfg.symbol}: ${kaufBetrag.toFixed(0)}€ (Topf ${cfg.capital_pool.toFixed(0)}€).` ); return { status: "buy", orderId: order.id, betrag: kaufBetrag, price: c.price, score: c.score }; } // 9) Verkaufen if (sellSig) { // Einstandswert der Position (was wurde investiert) vs. aktueller Marktwert const einstand = parseFloat(position!.cost_basis || "0"); const erloes = posValue; // aktueller Marktwert = Verkaufserlös const gewinn = erloes - einstand; // realisierter Gewinn/Verlust await alpaca(TRADING_BASE, "DELETE", `/v2/positions/${encodeURIComponent(cfg.symbol)}`); // --- Topf anpassen: Gewinn vergrößert ihn, Verlust verkleinert ihn --- const neuerPool = Math.max(0, cfg.capital_pool + gewinn); await supabase.from("bot_config").update({ capital_pool: neuerPool }).eq("id", 1); await logTrade({ symbol: cfg.symbol, side: "sell", notional: erloes, price: c.price, score: c.score, order_id: null, mode: "paper", }); await logStatus( `VERKAUF ${cfg.symbol}: Erlös ${erloes.toFixed(0)}€, ` + `G/V ${gewinn >= 0 ? "+" : ""}${gewinn.toFixed(0)}€ → Topf neu ${neuerPool.toFixed(0)}€.` ); return { status: "sell", price: c.price, gewinn, capital_pool: neuerPool }; } return { status: "hold", score: c.score, invested }; } // ---- HTTP-Einstiegspunkt (von Supabase-Scheduler aufgerufen) --------------- Deno.serve(async (req) => { if (req.method === "OPTIONS") { return new Response("ok", { headers: { "Access-Control-Allow-Origin": "*" } }); } try { const result = await runBotTick(); return new Response(JSON.stringify(result), { headers: { "Content-Type": "application/json" }, }); } catch (e) { const msg = e instanceof Error ? e.message : String(e); return new Response(JSON.stringify({ error: msg }), { status: 500, headers: { "Content-Type": "application/json" }, }); } });
